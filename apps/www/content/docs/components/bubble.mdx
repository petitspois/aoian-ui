## Usage

```tsx
import {
  Bubble,
  BubbleContent,
  BubbleWrapper,
} from "@/registry/default/ui/bubble"
```

```tsx
<Bubble>
  <BubbleWrapper>
    <BubbleContent>hello world !</BubbleContent>
  </BubbleWrapper>
</Bubble>
```
